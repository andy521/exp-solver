<script>
  export default {
    name: 'signIn',
    data() {
      return {
        username: '',
        password: ''
      }
    },
    methods: {
      signIn(){
        this.$api.authService.login({
          username: this.username,
          password: this.password
        }).then((rsp) => {
          this.$auth.login();
          this.$router.push("/homePage");
        }).catch((e) => {
          alert("账号密码错误，请重新输入");
        })
      },
      signUp(){
        this.$router.push("/login/signUp");
      },
      goBack(){
        this.$router.go(-1);
      }
    }
  }
</script>
